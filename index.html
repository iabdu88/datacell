// 1. دالة بدء التنظيف (التي يستدعيها الزر) function startCleaning() { if (!rawData || rawData.length === 0) { alert("لا توجد بيانات لتنظيفها!"); return; } // إظهار شاشة المعالجة showScreen('cleaning'); let count = 0; const total = rawData.length; const counterEl = document.getElementById('cleaning-counter'); // تأثير عداد الأرقام (أنيميشن) const interval = setInterval(() => { if (count >= total) { clearInterval(interval); executeFinalClean(); // تنفيذ التنظيف الفعلي بعد انتهاء الأنميشن } else { count += Math.ceil(total / 10); // زيادة سرعة العداد قليلاً if (count > total) count = total; if (counterEl) counterEl.innerText = count; } }, 40); } // 2. منطق التنظيف الفعلي (حل مشكلة الرموز + الدقة المحاسبية) function executeFinalClean() { cleanData = rawData.map(row => { let newRow = {}; for (let key in row) { let val = row[key]; if (typeof val === 'string') { // تنظيف الفراغات val = val.trim(); // حماية الحروف العربية والانجليزية والأرقام وحذف "الرموز الغريبة" // هذا التعبير النمطي يحافظ على النصوص العربية واللاتينية والمسافات val = val.replace(/[^\u0600-\u06FFa-zA-Z0-9\s]/g, ''); } newRow[key] = val; } return newRow; }); // الانتقال لشاشة النتائج وعرض البيانات المنظفة showScreen('insights'); renderResultTables(); } // 3. عرض الجداول في شاشة النتائج (تصحيح وسوم HTML) function renderResultTables() { const beforeTable = document.getElementById('clean-table-before'); const afterTable = document.getElementById('clean-table-after'); if (beforeTable) renderTable(beforeTable, rawData.slice(0, 5)); if (afterTable) renderTable(afterTable, cleanData.slice(0, 5)); } function renderTable(tableEl, data) { if (!data || data.length === 0) return; const headers = Object.keys(data[0]); // بناء الهيكل الصحيح للجدول let html = ""; headers.forEach(h => html += `${h}`); html += ""; data.forEach(row => { html += ""; headers.forEach(h => html += `${row[h]}`); html += ""; }); html += ""; tableEl.innerHTML = html; } // 4. دالة التصدير (Export) function exportCSV() { if (!cleanData || cleanData.length === 0) return; const ws = XLSX.utils.json_to_sheet(cleanData); const wb = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb, ws, "DataCell_Clean"); XLSX.writeFile(wb, "DataCell_Cleaned.xlsx"); } // هام جداً: إذا كنت تستخدم type="module" في السكريبت، أضف هذه السطور لضمان رؤية الأزرار للدوال window.startCleaning = startCleaning; window.exportCSV = exportCSV;

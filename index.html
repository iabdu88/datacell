// ═══════════════════════════════════════════════════════ // STATE // ═══════════════════════════════════════════════════════ let lang = 'ar'; let rawData = []; let cleanData = []; let headers = []; let rawHeaders = []; let filename = ''; let problems = { duplicates: 0, emptyCells: 0, outliers: 0, emptyColumns: 0 }; const EMPTY_VALS = new Set([ '', 'null', 'undefined', 'n/a', 'na', 'none', 'nil', 'missing', '?', '؟', 'لا يوجد', 'لا توجد', 'غير متاح', '-', '--', '#n/a', 'nan' ]); // ═══════════════════════════════════════════════════════ // FILE HANDLING // ═══════════════════════════════════════════════════════ function handleFile(e) { const file = e.target.files[0]; if (file) processFile(file); } function handleDrop(e) { e.preventDefault(); const file = e.dataTransfer.files[0]; if (file) processFile(file); } function processFile(file) { filename = file.name; const reader = new FileReader(); reader.onload = (e) => { try { const data = new Uint8Array(e.target.result); const wb = XLSX.read(data, { type: 'array' }); const ws = wb.Sheets[wb.SheetNames[0]]; const json = XLSX.utils.sheet_to_json(ws, { defval: '', raw: false }); if (!json || json.length === 0) { alert("الملف فارغ أو غير صالحة بياناته"); return; } rawData = json; headers = Object.keys(json[0]); rawHeaders = [...headers]; startProcessing(); } catch(err) { alert("خطأ في قراءة ملف الإكسل"); } }; reader.readAsArrayBuffer(file); } function startProcessing() { showScreen('processing'); const bar = document.getElementById('progress-bar'); if (bar) bar.style.width = '100%'; setTimeout(() => { analyzeData(); showDiagnosis(); }, 1500); } // ═══════════════════════════════════════════════════════ // ANALYSIS (منطق اكتشاف الأخطاء الحقيقي) // ═══════════════════════════════════════════════════════ function isEmpty(v) { if (v === null || v === undefined) return true; return EMPTY_VALS.has(String(v).trim().toLowerCase()); } function analyzeData() { problems = { duplicates: 0, emptyCells: 0, outliers: 0, emptyColumns: 0 }; const seenRows = new Set(); headers.forEach(col => { // 1. كشف الأعمدة الفارغة تماماً if (rawData.every(r => isEmpty(r[col]))) problems.emptyColumns++; // 2. كشف القيم الشاذة (إحصائياً) let nums = rawData.map(r => parseFloat(String(r[col]).replace(/,/g,''))).filter(n => !isNaN(n)); if (nums.length > 5) { let avg = nums.reduce((a,b) => a+b) / nums.length; let std = Math.sqrt(nums.map(x => Math.pow(x - avg, 2)).reduce((a,b) => a+b) / nums.length); rawData.forEach(r => { let n = parseFloat(String(r[col]).replace(/,/g,'')); if (!isNaN(n) && Math.abs(n - avg) > std * 2.5) problems.outliers++; }); } }); rawData.forEach(row => { // 3. كشف المكررات let rowKey = JSON.stringify(row); if (seenRows.has(rowKey)) problems.duplicates++; else seenRows.add(rowKey); // 4. كشف الفراغات headers.forEach(h => { if (isEmpty(row[h])) problems.emptyCells++; }); }); } // ═══════════════════════════════════════════════════════ // DIAGNOSIS (تحديث واجهة التقرير) // ═══════════════════════════════════════════════════════ function showDiagnosis() { showScreen('diagnosis'); // تحديث عدادات المشاكل في الواجهة const badges = document.querySelectorAll('.problem-badge span'); if (badges.length >= 2) { badges[0].textContent = (lang === 'ar' ? `خلايا فارغة` : `Empty Cells`) + ` (${problems.emptyCells})`; badges[1].textContent = (lang === 'ar' ? `صفوف مكررة` : `Duplicates`) + ` (${problems.duplicates})`; } // رندر الجدول الصغير للمعاينة if (typeof renderPreviewTable === 'function') renderPreviewTable(); } // ═══════════════════════════════════════════════════════ // CLEANING (منطق التنظيف الفعلي) // ═══════════════════════════════════════════════════════ function startCleaning() { showScreen('cleaning'); setTimeout(() => { const seen = new Set(); // تنفيذ التنظيف: حذف مكررات + مسح مسافات cleanData = rawData.filter(row => { let key = JSON.stringify(row); if (seen.has(key)) return false; seen.add(key); return true; }).map(row => { let newRow = {}; Object.keys(row).forEach(k => { newRow[k] = typeof row[k] === 'string' ? row[k].trim() : row[k]; }); return newRow; }); showInsights(); }, 2000); } // ═══════════════════════════════════════════════════════ // INSIGHTS & COMPARISON (إصلاح الرموز والعرض) // ═══════════════════════════════════════════════════════ function renderComparisonTable(tableId, data) { const tbl = document.getElementById(tableId); if (!tbl) return; const preview = data.slice(0, 10); const ths = `${rawHeaders.map(h => `${h}`).join('')}`; const trs = preview.map(row => { const tds = rawHeaders.map(col => { let val = (row[col] === null || row[col] === undefined) ? '' : String(row[col]); return `${val}`; }).join(''); return `${tds}`; }).join(''); tbl.innerHTML = `${ths}${trs}`; } function showInsights() { showScreen('insights'); renderComparisonTable('clean-table-before', rawData); renderComparisonTable('clean-table-after', cleanData); // تحديث عدادات قبل وبعد const bCount = document.getElementById('before-row-count'); const aCount = document.getElementById('after-row-count'); if (bCount) bCount.textContent = `(${rawData.length})`; if (aCount) aCount.textContent = `(${cleanData.length})`; } function showScreen(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); const target = document.getElementById('screen-' + id); if (target) target.classList.add('active'); } // ═══════════════════════════════════════════════════════ // GLOBAL EXPORTS // ═══════════════════════════════════════════════════════ window.handleFile = handleFile; window.handleDrop = handleDrop; window.startCleaning = startCleaning; window.showInsights = showInsights; window.loadSampleData = () => { alert("يرجى رفع ملف Excel حقيقي لتجربة التنظيف"); };

// 1. دالة بدء التنظيف (التي يستدعيها الزر) function startCleaning() { // إظهار شاشة المعالجة showScreen('cleaning'); let count = 0; const total = rawData.length; const counterEl = document.getElementById('cleaning-counter'); // تأثير عداد الأرقام (أنيميشن) const interval = setInterval(() => { if (count >= total) { clearInterval(interval); executeFinalClean(); // تنفيذ التنظيف الفعلي بعد انتهاء الأنميشن } else { count += Math.ceil(total / 20); // سرعة العداد if (count > total) count = total; counterEl.innerText = count; } }, 50); } // 2. منطق التنظيف الفعلي (الدقة المحاسبية) function executeFinalClean() { cleanData = rawData.map(row => { let newRow = {}; for (let key in row) { let val = row[key]; if (typeof val === 'string') { // تنظيف الفراغات + توحيد التنسيق + حل مشكلة الرموز val = val.trim(); // كود لمنع الرموز الغريبة مع الحفاظ على الحروف العربية والانجليزية والأرقام val = val.replace(/[^\u0600-\u06FFa-zA-Z0-9\s]/g, ''); } newRow[key] = val; } return newRow; }); // الانتقال لشاشة النتائج وعرض البيانات المنظفة showScreen('insights'); renderResultTables(); } // 3. عرض الجداول في شاشة النتائج function renderResultTables() { const beforeTable = document.getElementById('clean-table-before'); const afterTable = document.getElementById('clean-table-after'); // عرض أول 5 صفوف للمقارنة renderTable(beforeTable, rawData.slice(0, 5)); renderTable(afterTable, cleanData.slice(0, 5)); } function renderTable(tableEl, data) { if (!data || data.length === 0) return; const headers = Object.keys(data[0]); let html = `${headers.map(h => `${h}`).join('')}`; html += `${data.map(row => `${headers.map(h => `${row[h]}`).join('')}`).join('')}`; tableEl.innerHTML = html; } // 4. دالة التصدير (Export) لكي يعمل زر التحميل في النهاية function exportCSV() { const ws = XLSX.utils.json_to_sheet(cleanData); const wb = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb, ws, "CleanedData"); XLSX.writeFile(wb, "DataCell_Cleaned.xlsx"); }
